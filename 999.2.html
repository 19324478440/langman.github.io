Python 简易计算器

def calculate(num1, num2, op):
    """简单计算器函数：支持加减乘除"""
    if op == '+':
        return num1 + num2
    elif op == '-':
        return num1 - num2
    elif op == '*':
        return num1 * num2
    elif op == '/':
        return num1 / num2 if num2 != 0 else "除数不能为0"
    else:
        return "不支持的运算符号"

# 测试案例
print(calculate(10, 5, '+'))  # 输出：15
print(calculate(20, 4, '/'))  # 输出：5.0
print(calculate(7, 3, '*'))  # 输出：21

C++ 学生信息结构体

#include <iostream>
#include <string>
using namespace std;

// 定义学生结构体
struct Student {
    string name;  // 姓名
    int age;      // 年龄
    float score;  // 成绩
};

int main() {
    // 创建学生对象并赋值
    Student stu1 = {"张三", 20, 95.5};
    
    // 输出学生信息
    cout << "姓名：" << stu1.name << endl;
    cout << "年龄：" << stu1.age << endl;
    cout << "成绩：" << stu1.score << endl;
    
    return 0;
}

// 输出结果：
// 姓名：张三
// 年龄：20
// 成绩：95.5

Python 折线图绘制

import matplotlib.pyplot as plt
import numpy as np

# 生成数据：x轴为月份，y轴为销量
x = np.arange(1, 13)  # 1-12月
y = [20, 35, 30, 45, 50, 60, 75, 70, 65, 55, 40, 30]

# 设置中文显示
plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

# 绘制折线图
plt.plot(x, y, marker='o', color='blue', linewidth=2, markersize=6)
plt.title('月度销量趋势图', fontsize=14)
plt.xlabel('月份', fontsize=12)
plt.ylabel('销量（单位：件）', fontsize=12)
plt.xticks(x)  # x轴显示1-12
plt.grid(True, alpha=0.3)  # 添加网格
plt.show()
